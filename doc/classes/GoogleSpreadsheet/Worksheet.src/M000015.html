        def reload()
          doc = @session.get(@cells_feed_url)
          @cells = {}
          for entry in doc.search("entry")
            row = entry.search("gs:cell")[0]["row"].to_i()
            col = entry.search("gs:cell")[0]["col"].to_i()
            content = entry.search("content").text
            @cells[[row, col]] = content
          end
          @modified.clear()
          return true
        end

        def add_worksheet(title, max_rows = 100, max_cols = 20)
          xml = "<entry xmlns='http://www.w3.org/2005/Atom'\nxmlns:gs='http://schemas.google.com/spreadsheets/2006'>\n<title>\#{h(title)}</title>\n<gs:rowCount>\#{h(max_rows)}</gs:rowCount>\n<gs:colCount>\#{h(max_cols)}</gs:colCount>\n</entry>\n"
          doc = @session.post(@worksheets_feed_url, xml)
          url = doc.search(
            "link[@rel='http://schemas.google.com/spreadsheets/2006#cellsfeed']")[0]["href"]
          return Worksheet.new(@session, url, title)
        end
